@(matchState: cricket.CricketMatch) @main("Cricket Menu") {
@for(team <- matchState.teams) {

@team.name

@for(inns <- team.innings) {
@{s"${inns.runs}-${inns.wickets}"}
}
}
@matchState.teams(matchState.battingTeam).name @{s"${matchState.teams(matchState.battingTeam).innings(matchState.whichInnings).runs}-${matchState.teams(matchState.battingTeam).innings(matchState.whichInnings).wickets}"} @{s"(${matchState.teams(matchState.battingTeam).innings(matchState.whichInnings).overs}-${matchState.teams(matchState.battingTeam).innings(matchState.whichInnings).balls})"}
@matchState.teams(matchState.battingTeam).players(matchState.teams(matchState.battingTeam).innings(0).onStrike).name
@matchState.teams(matchState.battingTeam).players(matchState.teams(matchState.battingTeam).innings(0).waiting).name
}
